captcha_text = 'Какая версия языка сейчас актуальна?'
expected_result = 'Python3'
while True:
    user_answer = input('Какая версия языка сейчас актуальна?: '.format(captcha_text))
    if user_answer.title() == expected_result:
        print('Ответ {Python3} верен')
        score_true_try += 1
        break
    else:
        print('Неверный ответ.')
        score_false_try += 1
